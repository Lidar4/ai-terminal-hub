# Simple Socket Connection Example
# Educational Purpose: Demonstrates basic socket usage in Python

import socket

def create_connection(host, port):
    try:
        s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        s.settimeout(5) # Timeout after 5 seconds
        result = s.connect_ex((host, port))
        if result == 0:
            print(f"[+] Port {port} is open on {host}")
        else:
            print(f"[-] Port {port} is closed or filtered on {host}")
        s.close()
    except socket.gaierror:
        print(f"[!] Hostname '{host}' could not be resolved.")
    except socket.error as err:
        print(f"[!] Socket error: {err}")
